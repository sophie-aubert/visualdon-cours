Charger les données .json(), .csv()

Jusqu'à maintenant

Données = Pixels

Dès aujourd'hui

Données = Echelle * Pixels

Echelles

continu -> continu

Longuers, positions

continu -> discret

discret -> discret

.domain() & .range()

Exemple


		...
svg.append("circle")
	.attr("cx", 10)
	.attr("cy", 100)
	.attr("r", 40)
	.style("fill", "blue");

svg.append("circle")
	.attr("cx", 50)
	.attr("cy", 100)
	.attr("r", 40)
	.style("fill", "red");
		...
						
 
const x = d3.scaleLinear()
			.domain([0, 100])
			.range([0, 400]);
 
svg.append("circle")
		.attr("cx", x(10))
		.attr("cy", x(100))
		.attr("r", 40)
		.style("fill", "blue");

const margin = {top : 10, right: 40, bottom: 10, left: 40},
		   width = 450 - margin.left - margin.right,
		   height = 400 - margin.top - margin.bottom;
 
svg.attr("width", width + margin.left + margin.right)
    .attr("height", height + margin.top + margin.bottom)
  	.append("g")
    .attr("transform", "translate(" + margin.left + "," + margin.top + ")");
 
const x = d3.scaleLinear()
		.domain([0,100])
		.range([0,width])
 
svg.append('g')
  .attr("transform", "translate(0," + height + ")")
  .call(d3.axisBottom(x));
 
const y = d3.scaleLinear()
		.domain([100,0])
		.range([height,0])
 
svg.append('g')
  .call(d3.axisLeft(y));
